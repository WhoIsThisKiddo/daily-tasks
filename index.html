#!/usr/bin/env bash
# List the top 5 largest files in the current directory tree
du -ah . 2>/dev/null \
  | sort -rh \
  | head -n 5
Ez a script a következőket használja: a du az összegzett fájlméretekhez, a sort rendezéshez (humán olvasható, visszafelé), és a head a felső N sor kiválasztásához. A cél: gyorsan megtalálni a legnagyobb fájlokat a fájlrendszerben.


# Get top 5 processes by working set (memory)

  Sort-Object -Property WS -Descending |
  Select-Object -First 5 -Property Id,ProcessName,WS
