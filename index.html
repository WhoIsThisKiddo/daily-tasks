#!/usr/bin/env bash
set -euo pipefail

LOG="${1:-/var/log/nginx/access.log}"

# extract client IPs, count, sort by frequency, show top 10
awk '{print $1}' "$LOG" \
  | sort \
  | uniq -c \
  | sort -rn \
  | head -n 10

A script használata: paraméterként megadható a naplófájl útvonala, egyébként az alapértelmezett Nginx access.log lesz használva. Használt parancsok/szerkezetek: shebang, set -euo pipefail (hibakezelés), változóparaméterezés, awk a mezőkinyeréshez, sort/uniq -c a számláláshoz, sort -rn a gyakoriság szerinti rendezéshez, head a legfelső eredményekhez. A script célja: a legtöbbször előforduló kliens IP-címek gyors listázása.

Param(
    [string]$LogPath = "C:\inetpub\logs\LogFiles\W3SVC1\u_ex*.log"
)

# extract IPv4 addresses from IIS logs and show top 10
Get-ChildItem -Path $LogPath -File -ErrorAction SilentlyContinue |
    Select-String -Pattern '\b\d{1,3}(\.\d{1,3}){3}\b' -AllMatches |
    ForEach-Object { foreach ($m in $_.Matches) { $m.Value } } |
    Group-Object |
    Sort-Object Count -Descending |
    Select-Object -First 10 @{Name='IP';Expression={$_.Name}}, Count

Használt PowerShell elemek: Param blokkok a bemeneti paraméterhez, Get-ChildItem fájlok gyűjtésére, Select-String regex kereséshez (-AllMatches), ForEach-Object a találatok kibontására, Group-Object az összesítéshez, Sort-Object és Select-Object a rendezéshez és kiválasztáshoz. A szkript célja: IIS naplókban található IPv4 címek összegyűjtése és a leggyakoribb IP-k megjelenítése.
